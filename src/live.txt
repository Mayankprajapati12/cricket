                        }
                      }
                      else if (matchList.matchInfo.matchFormat !== "TEST") {
                        if (matchList.matchScore !== undefined) {
                          if (matchList.matchScore.team2Score !== undefined) {
                            return (
                              <>
                                <div className='h-36 w-11/12 bg-orange-500'>
                                {matchList.matchInfo.seriesName}<br />
                                venue : {matchList.matchInfo.venueInfo.ground} , {matchList.matchInfo.venueInfo.city}<br />
                                {matchList.matchInfo.team1.teamName}: {matchList.matchScore.team1Score.inngs1.runs} - {matchList.matchScore.team1Score.inngs1.wickets} ({matchList.matchScore.team1Score.inngs1.overs})<br />
                                {matchList.matchInfo.team2.teamName} : {matchList.matchScore.team2Score.inngs1.runs} - {matchList.matchScore.team2Score.inngs1.wickets} ({matchList.matchScore.team2Score.inngs1.overs})<br />
                                {matchList.matchInfo.status}<br />
                                <button onClick={() => { dispatch(idGen(matchList.matchInfo.matchId)) }}><Link to="/scorecard">show score</Link></button><hr />
                                </div>
                              </>
                            )
                          }
                          else if (matchList.matchScore.team1Score !== undefined) {
                            return (
                              <>
                                {matchList.matchInfo.seriesName}<br />
                                venue : {matchList.matchInfo.venueInfo.ground} , {matchList.matchInfo.venueInfo.city}<br />
                                {matchList.matchInfo.team1.teamName}: {matchList.matchScore.team1Score.inngs1.runs} - {matchList.matchScore.team1Score.inngs1.wickets} ({matchList.matchScore.team1Score.inngs1.overs})<br />
                                {matchList.matchInfo.status}<br />
                                <button onClick={() => { dispatch(idGen(matchList.matchInfo.matchId)) }}><Link to="/scorecard">show score</Link></button><hr />
                              </>
                            )
                          }
                        }
                        else {
                          return null
                        }
                      }
                      else {
                        return (
                          <>
                            {matchList.matchInfo.seriesName}<br />
                            venue : {matchList.matchInfo.venueInfo.ground} , {matchList.matchInfo.venueInfo.city}<br />
                            {matchList.matchInfo.status}<br />
                            <button onClick={() => { dispatch(idGen(matchList.matchInfo.matchId)) }}><Link to="/scorecard">show score</Link></button><hr />
                          </>
                        )
                      }
                    })
                  )
                }
                else {
                  return null
                }
              })
            )
          }))
      }
      else {
        return <>no live matches now</>
      }
    }
    else {
      return <>no live matches now</>
    }
  }
  else if (err !== 0) {
    return <>error:{err.message}</>
  }
  else {
    return <>wait....</>
  }
}
